  <!-- 載入 React 與 ReactDOM -->

  <script src="https://unpkg.com/react@18/umd/react.production.min.js" crossorigin></script>

  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

  <div id="root"></div>


  <script type="text/babel">

    const { useState, useMemo, useRef, useEffect } = React;


    // 預設的測試數據

    const defaultData = ` Look  Smile Talk  Touch

Look  -1.20 2.55  0.50  -0.30

Smile 0.10  -0.50 3.21  1.15

Talk  2.10  0.00  -1.10 2.85

Touch 0.50  1.50  -0.20 -0.80`;


    function App() {

      const [inputText, setInputText] = useState(defaultData);

      const [threshold, setThreshold] = useState(1.96);

      const [error, setError] = useState('');

      

      // 篩選狀態管理

      const [valueFilter, setValueFilter] = useState('all'); 

      const [visibleNodes, setVisibleNodes] = useState(new Set());

      

      // 拖曳狀態管理 (節點)

      const svgRef = useRef(null);

      const [manualPositions, setManualPositions] = useState({});

      const [draggingNode, setDraggingNode] = useState(null);

      

      // 拖曳狀態管理 (數值標籤)

      const [labelOffsets, setLabelOffsets] = useState({}); 

      const [draggingLabel, setDraggingLabel] = useState(null);


      // 版權與版本紀錄隱藏狀態

      const [showAbout, setShowAbout] = useState(false);

      

      // 畫布基本設定

      const canvasSize = 800;

      const canvasCenter = canvasSize / 2;


      // 解析輸入的文字資料 (支援 Tab、逗號、多個空白、特殊空白符號)

      const parsedData = useMemo(() => {

        try {

          setError('');

          if (!inputText.trim()) return { nodes: [], edges: [] };


          const lines = inputText.split('\n');

          if (lines.length < 2) throw new Error("資料行數不足，請提供包含標題的矩陣資料。");


          // 讀取並清理標題列 (替換 non-breaking space)

          const cleanHeaderLine = lines[0].replace(/\u00A0/g, ' ').trim();

          const headers = cleanHeaderLine.split(/\t|,|\s{2,}/).map(s => s.trim()).filter(s => s !== '');

          

          if (headers.length === 0) throw new Error("無法解析行為標題，請確認格式。");


          const nodesSet = new Set(headers);

          const edges = [];


          // 讀取數據列

          for (let i = 1; i < lines.length; i++) {

            const cleanLine = lines[i].replace(/\u00A0/g, ' ').trim();

            if (!cleanLine) continue;


            const columns = cleanLine.split(/\t|,|\s{2,}/).map(s => s.trim()).filter(s => s !== '');

            if (columns.length === 0) continue;


            const sourceNode = columns[0];

            nodesSet.add(sourceNode);


            for (let j = 1; j < columns.length; j++) {

              const targetNode = headers[j - 1];

              const value = parseFloat(columns[j]);

              

              if (!targetNode) continue;

              

              if (!isNaN(value)) {

                edges.push({

                  source: sourceNode,

                  target: targetNode,

                  value: value

                });

              }

            }

          }


          return {

            nodes: Array.from(nodesSet),

            edges: edges

          };

        } catch (err) {

          setError(err.message || '資料解析失敗，請確保格式為從 Excel 複製的表格。');

          return { nodes: [], edges: [] };

        }

      }, [inputText]);


      // 動態計算節點半徑：優化大小，設定合理的上限與下限，避免過度膨脹

      const nodeRadius = useMemo(() => {

        if (parsedData.nodes.length === 0) return 32;

        const maxLen = Math.max(...parsedData.nodes.map(n => n.length));

        // 基礎大小 32，字元越多稍微放大，但最高不超過 42

        return Math.min(42, Math.max(32, maxLen * 2.8));

      }, [parsedData.nodes]);


      // 當資料改變時，重置拖曳位置

      useEffect(() => {

        setManualPositions({});

        setLabelOffsets({}); 

        setVisibleNodes(new Set(parsedData.nodes));

      }, [parsedData.nodes]);


      // 自動排版與對齊功能

      const handleFormatText = () => {

        try {

          const lines = inputText.split('\n');

          if (lines.length < 2) return;


          const headers = lines[0].replace(/\u00A0/g, ' ').trim().split(/\t|,|\s{2,}/).map(s => s.trim()).filter(s => s !== '');

          if (headers.length === 0) return;


          const rows = [ ["", ...headers] ]; // 第一列左上角留空


          for (let i = 1; i < lines.length; i++) {

            const cleanLine = lines[i].replace(/\u00A0/g, ' ').trim();

            if (!cleanLine) continue;

            const columns = cleanLine.split(/\t|,|\s{2,}/).map(s => s.trim()).filter(s => s !== '');

            if (columns.length > 0) {

              rows.push(columns);

            }

          }


          // 計算每個欄位的最大寬度

          const colWidths = [];

          rows.forEach(row => {

            row.forEach((cell, i) => {

              colWidths[i] = Math.max(colWidths[i] || 0, cell.length);

            });

          });


          // 重建對齊後的文字

          const formattedText = rows.map(row => {

            return row.map((cell, i) => cell.padEnd(colWidths[i] + 3, ' ')).join('');

          }).join('\n');


          setInputText(formattedText);

        } catch (err) {

          console.error("Formatting failed", err);

          setError("文字排版失敗，請確認資料格式是否正確。");

        }

      };


      // 節點篩選切換邏輯

      const toggleNode = (node) => {

        setVisibleNodes(prev => {

          const next = new Set(prev);

          if (next.has(node)) next.delete(node);

          else next.add(node);

          return next;

        });

      };


      const toggleAllNodes = () => {

        if (visibleNodes.size === parsedData.nodes.length) {

          setVisibleNodes(new Set()); 

        } else {

          setVisibleNodes(new Set(parsedData.nodes)); 

        }

      };


      // 過濾連線

      const filteredEdges = useMemo(() => {

        return parsedData.edges.filter(edge => {

          if (!visibleNodes.has(edge.source) || !visibleNodes.has(edge.target)) return false;

          if (Math.abs(edge.value) < threshold) return false;

          if (valueFilter === 'positive' && edge.value < 0) return false;

          if (valueFilter === 'negative' && edge.value >= 0) return false;

          return true;

        });

      }, [parsedData.edges, threshold, visibleNodes, valueFilter]);


      // 計算預設座標

      const defaultPositions = useMemo(() => {

        const radius = 300;

        const pos = {};

        parsedData.nodes.forEach((node, index) => {

          const angle = (index / parsedData.nodes.length) * 2 * Math.PI - Math.PI / 2; 

          pos[node] = {

            x: canvasCenter + radius * Math.cos(angle),

            y: canvasCenter + radius * Math.sin(angle),

          };

        });

        return pos;

      }, [parsedData.nodes, canvasCenter]);


      const currentPositions = useMemo(() => {

        return { ...defaultPositions, ...manualPositions };

      }, [defaultPositions, manualPositions]);


      // --- 拖曳事件處理 ---

      const handlePointerDown = (e, node) => {

        e.preventDefault();

        e.stopPropagation();

        setDraggingNode(node);

        e.target.setPointerCapture(e.pointerId); 

      };


      const handleLabelPointerDown = (e, edgeKey) => {

        e.preventDefault();

        e.stopPropagation();

        

        const svg = svgRef.current;

        const pt = svg.createSVGPoint();

        pt.x = e.clientX;

        pt.y = e.clientY;

        const cursorPt = pt.matrixTransform(svg.getScreenCTM().inverse());

        

        setDraggingLabel({

          key: edgeKey,

          startX: cursorPt.x,

          startY: cursorPt.y,

          initOffsetX: labelOffsets[edgeKey]?.x || 0,

          initOffsetY: labelOffsets[edgeKey]?.y || 0

        });

        

        e.target.setPointerCapture(e.pointerId);

      };


      const handlePointerMove = (e) => {

        if (!svgRef.current) return;

        

        const svg = svgRef.current;

        const pt = svg.createSVGPoint();

        pt.x = e.clientX;

        pt.y = e.clientY;

        const cursorPt = pt.matrixTransform(svg.getScreenCTM().inverse());


        if (draggingNode) {

          setManualPositions(prev => ({

            ...prev,

            [draggingNode]: { x: cursorPt.x, y: cursorPt.y }

          }));

        } 

        else if (draggingLabel) {

          const dx = cursorPt.x - draggingLabel.startX;

          const dy = cursorPt.y - draggingLabel.startY;

          

          setLabelOffsets(prev => ({

            ...prev,

            [draggingLabel.key]: {

              x: draggingLabel.initOffsetX + dx,

              y: draggingLabel.initOffsetY + dy

            }

          }));

        }

      };


      const handlePointerUp = (e) => {

        if (draggingNode) {

          setDraggingNode(null);

          if (e.target.hasPointerCapture(e.pointerId)) e.target.releasePointerCapture(e.pointerId);

        }

        if (draggingLabel) {

          setDraggingLabel(null);

          if (e.target.hasPointerCapture(e.pointerId)) e.target.releasePointerCapture(e.pointerId);

        }

      };


      // 下載 PNG

      const handleDownloadPNG = () => {

        if (!svgRef.current) return;

        const svgElement = svgRef.current;

        const svgData = new XMLSerializer().serializeToString(svgElement);

        

        const canvas = document.createElement("canvas");

        const ctx = canvas.getContext("2d");

        const img = new Image();

        canvas.width = canvasSize;

        canvas.height = canvasSize;


        img.onload = () => {

          ctx.fillStyle = "white";

          ctx.fillRect(0, 0, canvas.width, canvas.height);

          ctx.drawImage(img, 0, 0);

          

          const pngFile = canvas.toDataURL("image/png");

          const downloadLink = document.createElement("a");

          downloadLink.download = "gseq-plot.png";

          downloadLink.href = pngFile;

          downloadLink.click();

        };


        img.src = "data:image/svg+xml;base64," + btoa(unescape(encodeURIComponent(svgData)));

      };


      const resetLayout = () => {

        setManualPositions({});

        setLabelOffsets({});

      };


      // 繪製路徑計算

      const calculateCurve = (source, target, value) => {

        const s = currentPositions[source];

        const t = currentPositions[target];

        if (!s || !t) return null;


        if (source === target) {

          let nx = s.x - canvasCenter;

          let ny = s.y - canvasCenter;

          const dist = Math.sqrt(nx*nx + ny*ny) || 1;

          nx = nx / dist;

          ny = ny / dist;

          if (s.x === canvasCenter && s.y === canvasCenter) { nx = 0; ny = -1; }


          const px = -ny;

          const py = nx;

          const finalSx = s.x + (nx * 0.866 + px * 0.5) * nodeRadius;

          const finalSy = s.y + (ny * 0.866 + py * 0.5) * nodeRadius;

          const finalTx = s.x + (nx * 0.866 - px * 0.5) * nodeRadius;

          const finalTy = s.y + (ny * 0.866 - py * 0.5) * nodeRadius;


          // 動態計算貝茲曲線的控制點距離

          const loopDist = nodeRadius * 2.5 + 25;

          const loopSpread = nodeRadius * 1.5 + 15;

          const cp1x = s.x + nx * loopDist + px * loopSpread;

          const cp1y = s.y + ny * loopDist + py * loopSpread;

          const cp2x = s.x + nx * loopDist - px * loopSpread;

          const cp2y = s.y + ny * loopDist - py * loopSpread;

          

          return {

            path: `M ${finalSx} ${finalSy} C ${cp1x} ${cp1y}, ${cp2x} ${cp2y}, ${finalTx} ${finalTy}`,

            textX: s.x + nx * (loopDist + 15),

            textY: s.y + ny * (loopDist + 15),

            isPositive: value >= 0

          };

        }


        const dx = t.x - s.x;

        const dy = t.y - s.y;

        const dist = Math.sqrt(dx * dx + dy * dy);

        if (dist === 0) return null;


        const ux = dx / dist;

        const uy = dy / dist;

        const offset = 12; 

        const nx = -uy;

        const ny = ux;

        

        const sx = s.x + nx * offset;

        const sy = s.y + ny * offset;

        const tx = t.x + nx * offset;

        const ty = t.y + ny * offset;


        const finalSx = sx + ux * nodeRadius;

        const finalSy = sy + uy * nodeRadius;

        const finalTx = tx - ux * nodeRadius;

        const finalTy = ty - uy * nodeRadius;


        const midX = (finalSx + finalTx) / 2;

        const midY = (finalSy + finalTy) / 2;


        return {

          path: `M ${finalSx} ${finalSy} L ${finalTx} ${finalTy}`,

          textX: midX,

          textY: midY,

          isPositive: value >= 0

        };

      };


      return (

        <div className="min-h-screen p-6 font-sans text-gray-800">

          <div className="max-w-[1400px] mx-auto space-y-6">

            

            <header className="bg-white p-6 rounded-xl shadow-sm border border-gray-200">

              <h1 className="text-2xl font-bold text-gray-800">GSEQ 序列分析繪圖工具</h1>

              <p className="text-gray-500 mt-2">

                將 GSEQ 矩陣報表貼上下方文字框。右側畫布支援<strong>自由拖曳節點</strong>與<strong>數值標籤</strong>，避免資料重疊。

              </p>

            </header>


            <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">

              

              {/* 左側控制面板 */}

              <div className="space-y-6 lg:col-span-3">

                

                {/* 1. 資料輸入 */}

                <div className="bg-white p-6 rounded-xl shadow-sm border border-gray-200">

                  <label className="block text-sm font-semibold text-gray-700 mb-2">

                    1. 貼上矩陣資料

                  </label>

                  <textarea

                    className="w-full min-h-[200px] h-64 lg:h-96 resize-y overflow-auto p-3 border border-gray-300 rounded-lg text-sm font-mono focus:ring-2 focus:ring-blue-500 focus:border-blue-500 outline-none"

                    value={inputText}

                    onChange={(e) => setInputText(e.target.value)}

                    placeholder="貼上你的 Excel 資料..."

                    wrap="off"

                  />

                  {error && <p className="text-red-500 text-sm mt-2 font-medium">{error}</p>}

                  

                  <div className="flex flex-col gap-2 mt-3">

                    <button 

                      onClick={handleFormatText}

                      className="w-full py-2 bg-indigo-50 hover:bg-indigo-100 text-indigo-700 border border-indigo-200 text-sm font-bold rounded-lg transition-colors shadow-sm"

                    >

                      ✨ 自動排版對齊 (整理貼上資料)

                    </button>

                    <div className="flex gap-2">

                      <button 

                        onClick={resetLayout}

                        className="flex-1 py-2 bg-gray-100 hover:bg-gray-200 text-gray-700 text-sm font-medium rounded-lg transition-colors"

                      >

                        重置畫布

                      </button>

                      <button 

                        onClick={handleDownloadPNG}

                        className="flex-1 py-2 bg-blue-600 hover:bg-blue-700 text-white text-sm font-medium rounded-lg shadow-sm transition-colors"

                      >

                        下載圖片

                      </button>

                    </div>

                  </div>

                </div>


                {/* 2. 顯著水準 */}

                <div className="bg-white p-6 rounded-xl shadow-sm border border-gray-200">

                  <label className="block text-sm font-semibold text-gray-700 mb-2">

                    2. 顯著水準閾值 (Z值)

                  </label>

                  <div className="flex items-center gap-4">

                    <input

                      type="range"

                      min="0"

                      max="5"

                      step="0.01"

                      value={threshold}

                      onChange={(e) => setThreshold(parseFloat(e.target.value))}

                      className="flex-1"

                    />

                    <span className="font-mono bg-gray-100 px-3 py-1 rounded text-sm font-semibold">

                      {threshold.toFixed(2)}

                    </span>

                  </div>

                </div>


                {/* 3. 篩選條件 */}

                <div className="bg-white p-6 rounded-xl shadow-sm border border-gray-200">

                  <label className="block text-sm font-semibold text-gray-700 mb-3">

                    3. 篩選條件

                  </label>

                  <div className="mb-4 pb-4 border-b border-gray-100">

                    <div className="flex flex-col gap-2 text-sm text-gray-700">

                      <label className="flex items-center gap-2 cursor-pointer">

                        <input type="radio" name="valueFilter" value="all" checked={valueFilter === 'all'} onChange={() => setValueFilter('all')} className="text-blue-600 focus:ring-blue-500" /> 全部顯示

                      </label>

                      <label className="flex items-center gap-2 cursor-pointer">

                        <input type="radio" name="valueFilter" value="positive" checked={valueFilter === 'positive'} onChange={() => setValueFilter('positive')} className="text-blue-600 focus:ring-blue-500" /> 僅顯示正向 (促進)

                      </label>

                      <label className="flex items-center gap-2 cursor-pointer">

                        <input type="radio" name="valueFilter" value="negative" checked={valueFilter === 'negative'} onChange={() => setValueFilter('negative')} className="text-red-500 focus:ring-red-500" /> 僅顯示負向 (抑制)

                      </label>

                    </div>

                  </div>

                  <div>

                    <div className="flex justify-between items-center mb-2">

                      <p className="text-xs text-gray-500 font-medium">顯示參與的節點：</p>

                      <button onClick={toggleAllNodes} className="text-xs font-medium text-blue-600 hover:text-blue-800 transition-colors">
